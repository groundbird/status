fig, ax = plt.subplots()
for i in range(8):
    ax.plot(tod.dates(10), tod.vals(10)[:,i], label='ch. '+str(i))
ax.xaxis.set_major_formatter(md.DateFormatter('%m/%d\n%H:%M', tz=pytz.timezone('Asia/Tokyo')))
ax.set_ylabel('Temperature [K]')
leg = ax.legend(loc='best')
leg.get_frame().set_alpha(0.5)
ax.grid()
plt.show()
